%%% Headers and page styles
\usepackage[pagestyles]{titlesec}
\usepackage{fancyhdr}
\setlength{\headheight}{15.2pt}

% ourbook style with fancy headers and chapter headings
\fancypagestyle{ourbook}{
  % headers
  \fancyhf{}
  \fancyhf[FRO,FLE]{\HeaderFont{\thepage}}
  \fancyhf[HLE]{\HeaderFont{\leftmark}}
  \fancyhf[HRO]{\HeaderFont{\AuthorFull}}
  \renewcommand{\headrulewidth}{0.5pt}
  \renewcommand{\printchaptername}{}
  \renewcommand{\chapternamenum}{}
  \renewcommand{\printchapternum}{}
  \renewcommand{\printchaptertitle}[1]{%
  \TitleFamily\huge ##1}
  \setlength{\parskip}{0pt}
}

% plain style with only page num
\fancypagestyle{plain}{
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
  \fancyhf[FRO,FLE]{\thepage}
  \renewcommand{\printchaptertitle}[1]{%
  \TitleFamily\huge ##1}
}

% single space after periods
\frenchspacing

% Attempt justification at all costs
\sloppy

% Widows and orphans
\widowpenalty=10000
\clubpenalty=10000

\newcommand\ygTimestamp[1]{
